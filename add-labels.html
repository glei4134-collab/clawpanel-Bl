给聊天区域添加可见标签

复制以下代码到浏览器控制台(F12)执行：

(function() {
  const areas = [
    { selector: ".chat-sidebar", label: "chat-sidebar" },
    { selector: ".chat-sidebar-header", label: "chat-sidebar-header" },
    { selector: ".chat-session-list", label: "chat-session-list" },
    { selector: ".chat-main", label: "chat-main" },
    { selector: ".chat-header", label: "chat-header" },
    { selector: ".chat-messages-area", label: "chat-messages-area" },
    { selector: ".chat-input-area", label: "chat-input-area" }
  ];
  
  areas.forEach(function(item) {
    var el = document.querySelector(item.selector);
    if (el) {
      var tag = document.createElement("div");
      tag.textContent = item.label;
      tag.style.cssText = "position:fixed;top:5px;left:5px;background:red;color:white;padding:2px 6px;font-size:12px;z-index:99999;pointer-events:none;opacity:0.9;";
      el.style.position = "relative";
      el.appendChild(tag);
    }
  });
  
  console.log("已添加标签到 " + areas.filter(function(a) { return document.querySelector(a.selector); }).length + " 个区域");
})();
